MySQL server running on Ubuntu subsystem for windows.
Installation instructions found here:
https://docs.microsoft.com/en-us/windows/wsl/install-win10

Run Ubuntu setup and set user and root password then do following commands:

Sudo apt update
sudo apt install mysql-server
sudo service mysql start
sudo mysql


CREATE DATABASE SalesScript;
use SalesScript
CREATE TABLE users(ID int auto_increment, username VARCHAR(15), fName VARCHAR(30), lName VARCHAR(30), email VARCHAR(60), password VARCHAR(50), PRIMARY KEY(ID));

CREATE USER 'Jack'@'localhost' IDENTIFIED BY 'password';
CREATE USER 'Ryan'@'localhost' IDENTIFIED BY 'password';
CREATE USER 'Jeanette'@'localhost' IDENTIFIED BY 'password';

GRANT ALL PRIVILEGES ON * . * TO 'Jack'@'localhost';
GRANT ALL PRIVILEGES ON * . * TO 'Ryan'@'localhost';
GRANT ALL PRIVILEGES ON * . * TO 'Jeanette'@'localhost';
